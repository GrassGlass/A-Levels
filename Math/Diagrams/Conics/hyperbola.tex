                    }}

% arrows as stealth fighters
\tikzset{>=stealth,
-<-/.style={decoration={
  markings,
  mark=at position #1 with {\arrow[scale=1.5]{Classical TikZ Rightarrow[reversed]}}},postaction={decorate}}}
\NewDocumentCommand\bisector{O{}mmmO{10pt}}{
\path[#1] let
    \p1 = ($(#3)!10cm!(#2)$),
    \p2 = ($(#3)!10cm!(#4)$),
    \p3 = ($(\p1) + (\p2) - (#3)$)
    in    
    ($(#3)!#5!($(#3)!0.85cm!(\p3)$)$) -- ($($(#3)!0.85cm!(\p3)$)!#5!(#3)$) ;
}
